df = pd.read_csv('./features2.csv')

f1_index = df.columns.get_loc("F1")


for index, row in df.iterrows():
    if row["F1"] == 1:

        for i in range(f1_index, len(df.columns) - 1):
            df.iloc[index, i] = df.iloc[index, i + 1]
        
        
        df.iloc[index, len(df.columns) - 1] = pd.NA


df.to_csv('./features2.csv', index=False)
